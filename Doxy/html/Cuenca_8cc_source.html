214int Cuenca::simular_comercio(Barco& barco, Ciudad& c) {
215 int compras = 0;
216 int ventas = 0;
217 int idc = barco.consultar_comprando();
218 int idv = barco.consultar_vendiendo();
219 if (c.existe_producto(idc)) {
220 int oferta = c.consultar_cantidad(idc) - c.consultar_necesarios(idc);
221 if (oferta > 0) {
222 if (barco.consultar_comprando_cantidad() > oferta) {
223 compras += oferta;
224 barco.restar_comprando(oferta);
225 }
226 else {
227 compras += barco.consultar_comprando_cantidad();
228 barco.modificar_comprando(idc, 0);
229 }
230 }
231 }
232 if (c.existe_producto(idv)) {
233 int demanda = c.consultar_necesarios(idv) - c.consultar_cantidad(idv);
234 if (demanda > 0) {
235 if (barco.consultar_vendiendo_cantidad() < demanda) {
236 ventas += barco.consultar_vendiendo_cantidad();
237 barco.modificar_vendiendo(idv, 0);
238 }
239 else {
240 ventas += demanda;
241 barco.restar_vendiendo(demanda);
242 }
243 }
244 }
245 return compras + ventas;
246}
